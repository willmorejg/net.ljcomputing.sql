017package net.ljcomputing.sql.clause;
018
019import net.ljcomputing.sql.expression.Expression;
020import net.ljcomputing.sql.identifier.Column;
021import net.ljcomputing.sql.literal.Literal;
022import net.ljcomputing.sql.literal.Operand;
023
024/**
025 * Abstract implementation of a predicate clause.
026 * 
027 * @author James G. Willmore
028 *
029 */
030public abstract class AbstractPredicate implements Predicate {
031  
032  /** The expression. */
033  private final transient Expression expression;
034  
035  /** The value. */
036  private final transient Object value;
037
038  /**
039   * Instantiates a new predicate.
040   *
041   * @param expression the expression
042   * @param value the value
043   */
044  public AbstractPredicate(final Expression expression, final Object value) {
045    this.expression = expression;
046    this.value = value;
047  }
048  
049  /**
050   * SQL predicate.
051   *
052   * @return the string
053   */
054  public final String toPredicate() {
055    final StringBuffer buf = new StringBuffer();
056    final Column column = expression.getColumn();
057    final String columnName = column.getName();
058    final String columnAlias = column.getAlias();
059    final Operand operand = expression.getOperand();
060    final String valueStr = value != null ? value.toString() : null;
061    
062    if(columnAlias != null) {
063      buf.append(columnAlias).append(Literal.Period);
064    }
065    
066    buf.append(columnName).append(operand.toString()).append(valueStr);
067    
068    return buf.toString();
069  }
070}
